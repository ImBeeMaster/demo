node {
   def server = Artifactory.server('Local')
   def mvnHome
   //def commitHash = checkout(scm).GIT_COMMIT.substring(0,8)
   // def commitBranch = checkout(scm).GIT_BRANCH.replace('origin/','')
   def uploadSpec = """{
  "files": [
    {
      "pattern": "target/*.jar",
      "target": "libs-release-local"
    }
 ]
}"""
      stage('Preparation') { // for display purposes
      // Get some code from a GitHub repository
     git branch: 'Dev', url: 'git@github.com:https://github.com/ImBeeMaster/demo.git'
     // checkout scm
      // comment 
      sh "echo this is: ${commitHash}"
      sh "echo this is: ${commitBranch}"

      // Get the Maven tool.
      // ** NOTE: This 'M3' Maven tool must be configured
      // **       in the global configuration.           
      mvnHome = tool 'Maven_build'
      
   }
   stage('Build') {
      // Run the maven build
      if (isUnix()) {
         // echo "Build happens here"
         sh "'${mvnHome}/bin/mvn' -Dmaven.test.failure.ignore clean package -Dv=${commitHash} -Db=${commitBranch}"

      } else {
         bat(/"${mvnHome}\bin\mvn" -Dmaven.test.failure.ignore clean package/)
      }
      server.upload(uploadSpec) // upload to registry
      sh "rm -rf ${env.WORKSPACE}/target/*.jar"
      }
}
