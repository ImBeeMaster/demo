node {
   def server = Artifactory.server('Local')
   def mvnHome
   def env.GIT_COMMIT
   def env.GIT_BRANCH
   def uploadSpec = """{
  "files": [
    {
      "pattern": "target/*.jar",
      "target": "libs-release-local"
    }
 ]
}"""
      stage('Preparation') { // for display purposes
      // Get some code from a GitHub repository
      git 'https://github.com/ImBeeMaster/demo.git'
      sh "echo ${GIT_COMMIT}"
      sh "echo ${GIT_BRANCH}"

      // Get the Maven tool.
      // ** NOTE: This 'M3' Maven tool must be configured
      // **       in the global configuration.           
      mvnHome = tool 'Maven_build'
      
   }
   stage('Build') {
      // Run the maven build
      if (isUnix()) {
         
         sh "'${mvnHome}/bin/mvn' -Dmaven.test.failure.ignore clean package -Dv=${env.GIT_COMMIT} -Db=${env.GIT_BRANCH}"

      } else {
         bat(/"${mvnHome}\bin\mvn" -Dmaven.test.failure.ignore clean package/)
      }
        sh "rm -rf ${env.WORKSPACE}/target/*.jar"
        server.upload(uploadSpec) // upload to registry
      }
}
